{extend name="template/base" /} {block name="content"}
``` layer_open('默认弹层','{:url(\'layer\')}'); //默认弹层宽高都为80% ```
``` layer_open('全屏弹层','{:url(\'layer\')}',{w:'100%',h:'100%'}); //全屏弹层宽高都为100% ```
``` layer_open('带警告的弹层','{:url(\'layer\')}',{confirm:true}); //弹层关闭前会有警告提醒 ```
``` layer_open('带回调的弹层','{:url(\'layer\')}',{fn:function() {layer.alert('我是回调，会在弹层结束后触发')}}); //回调会在弹层结束后触发 ```
``` layer_open('使用HTML内联弹层',$('#tpl').html(),{type:1}); //使用HTML内联弹层，可以把HTML定义在script标签里，把type定义为text/html ```
以上弹层的参数都可以组合使用
